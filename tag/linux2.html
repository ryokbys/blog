Automate git archive

I made a bash function to replace git archive command,

function gitarch(){
    basedname=$(git rev-parse --show-toplevel)
    dname=$(basename $basedname)
    cwd=$(pwd)
    cd $basedname
    fname=${dname}_`ymd`.tgz
    git archive --format=tar --prefix=${dname}/ HEAD | gzip > $fname
    cp $fname ${cwd}/
    cd $cwd
}

#!/bin/bash

now=`date +%y%m%d_%H%M%S`
dname=results_$now
mkdir $dname
cp some result files $dname/
echo some result files
echo "These files are Saved at " $dname
